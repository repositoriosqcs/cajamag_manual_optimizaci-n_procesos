    Campos Calculados¶

    En esta sección se presentan los campos calculados o columnas creadas en DAX para la construcción de la capa visual.

    1. EstadoDes

    Tabla: Historico

    
IF( Historico[estado] = "A", "Activo", 
    IF( OR( Historico[estado] = "I", Historico[estado] = "M"), "Inactivo", 
        IF( Historico[estado] = "P", "Pendiente", BLANK() )
    )
)
    2. KeyPromedioAporte

    Tabla: Historico

    
RIGHT( Historico[cod], LEN(Historico[cod]) - 2 ) & LEFT(Historico[Tipo], 3)
    3. RangoNumTrabajadores

    Tabla: Historico

    
IF( ISBLANK(Historico[NumTrab] ), BLANK(), 
    IF( Historico[NumTrab] <= 10, "0-10 trab", 
        IF( AND( 10 < Historico[NumTrab], Historico[NumTrab] <= 50 ), "11-50 trab", 
            IF( AND( 49 < Historico[NumTrab], Historico[NumTrab] <= 250 ), "51-250 trab", 
                IF( 250 < Historico[NumTrab], "<250 trab", BLANK() )
            )
        )
    )
)
    4. tipcaptra

    Tabla: Historico

    
IF( Historico[captra] = "N", "Normal", 
    IF( Historico[captra] = "I", "Discapacidad", Historico[captra])
)
    5. tipparent

    Tabla: Historico

    
IF( Historico[parent] = "1", "Hijo", 
    IF( Historico[parent] = "2", "Hermano", 
        IF( Historico[parent] = "3", "Padre")
    )
)
    6. c_ANS

    Tabla: HistoricoANS

    
IF(
    (HistoricoANS[Fuente] = "subsi15" && SELECTEDVALUE(Historico[tipafi]) <> 7 && SELECTEDVALUE(Historico[tipafi]) <> 8 
    && SELECTEDVALUE(Historico[tipafi]) <> 4 && SELECTEDVALUE(Historico[tipafi]) <> 5 && SELECTEDVALUE(Historico[tipafi]) <> 9 
    && SELECTEDVALUE(Historico[tipafi]) <> 11 && SELECTEDVALUE(Historico[tipafi]) <> 6 && HistoricoANS[DiasTrabajados] > 8)
    || (HistoricoANS[Fuente] = "subsi23" && HistoricoANS[DiasTrabajados] > 8),
    "Fuera de Ans",
    IF(
        (HistoricoANS[Fuente] = "subsi15" && SELECTEDVALUE(Historico[tipafi]) <> 7 && SELECTEDVALUE(Historico[tipafi]) <> 8 
        && HistoricoANS[DiasTrabajados] > 3)
        || (HistoricoANS[Fuente] = "subsi15" && SELECTEDVALUE(Historico[tipafi]) <> 4 && SELECTEDVALUE(Historico[tipafi]) <> 5 
        && SELECTEDVALUE(Historico[tipafi]) <> 9 && SELECTEDVALUE(Historico[tipafi]) <> 11 && HistoricoANS[DiasTrabajados] > 3)
        || (HistoricoANS[Fuente] = "subsi15" && SELECTEDVALUE(Historico[tipafi]) <> 6 && HistoricoANS[DiasTrabajados] > 3)
        || (HistoricoANS[Fuente] = "subsi02" && HistoricoANS[DiasTrabajados] > 3),
        "Fuera de Ans",
        "Dentro de Ans"
    )
)
    7. DiasTrabajados

    Tabla: HistoricoANS

    
NETWORKDAYS( HistoricoANS[fecafi], HistoricoANS[fecapr], 1, FestivosColombia)
    8. NoAfil

    Tabla: HistóricoAportesEmp

    
IF( LOOKUPVALUE( Historico[KeyMoraEmp], Historico[KeyMoraEmp], HistoricoAportesEmp[KeyHistoricos] ) = BLANK(), 1, 0)
    9. CumpleMeta

    Tabla: HistoricoCuota

    
IF( HistoricoCuota[fecent] <= HistoricoCuota[Meta], "Cumple", 
    IF( ISBLANK(HistoricoCuota[Meta]), BLANK(), "No Cumple")
)
    10. descuento

    Tabla: HistoricoCuota

    
IF(
    (HistoricoCuota[codcue] = "16" || HistoricoCuota[codcue] = "61" || HistoricoCuota[codcue] = "98"),
    "Si",
    "No"
)
    11. Meta

    Tabla: HistoricoCuota

    
VAR pergir = HistoricoCuota[NextPergir]
VAR gir = HistoricoCuota[numgir]
VAR dev = CALCULATE( MAX(MetasCuota[fecha]), MetasCuota[periodo] = pergir, MetasCuota[numGiro] = gir)
RETURN dev
    12. NextPergir

    Tabla: HistoricoCuota

    
VAR pergir = HistoricoCuota[periodo]
VAR anio = VALUE(LEFT(pergir, 4))
VAR mes = VALUE(RIGHT(pergir, 2))
VAR dev = IF( mes = 12, (anio+1)*100 + 1, anio*100 + (mes+1))

RETURN CONVERT(dev, STRING)
    13. DiasMora

    Tabla: HistoricoMoraEmp

    
VAR nit = HistoricoMoraEmp[nit]
VAR periodo = HistoricoMoraEmp[periodo_actualizacion]
VAR fech_Comp = EOMONTH(EDATE(TODAY(), -2), 0)
VAR PerAnt = CALCULATE( MAX(HistoricoMoraEmp[periodo_actualizacion]), ALL(HistoricoMoraEmp), HistoricoMoraEmp[Primero] = 1, periodo >= HistoricoMoraEmp[periodo_actualizacion], nit = HistoricoMoraEmp[nit])
RETURN DATEDIFF(DATEVALUE("01/" & RIGHT(PerAnt, 2) & "/" & LEFT(PerAnt, 4)), fech_Comp, DAY)
    14. EdadesPresuntas

    Tabla: HistoricoMoraEmp

    
IF( AND(HistoricoMoraEmp[DiasMora] >= 30, HistoricoMoraEmp[DiasMora] <= 60), "30-60 días",
    IF( AND(HistoricoMoraEmp[DiasMora] > 60, HistoricoMoraEmp[DiasMora] <= 90), "61-90 días",
        IF( HistoricoMoraEmp[DiasMora] > 90, "> 91 días", BLANK())
    )
)
    15. NoConvenio

    Tabla: HistoricoMoraEmp

    
IF( LOOKUPVALUE(HistoricoRecuperado[KeyXML4], HistoricoRecuperado[KeyXML4], HistoricoMoraEmp[KeyConvenio]) = BLANK(), 1, 0)
    16. PeriodoAntiguo

    Tabla: HistoricoMoraEmp

    
VAR nit = HistoricoMoraEmp[nit]
VAR periodo = HistoricoMoraEmp[periodo_actualizacion]
VAR periodo_antig = CALCULATE(MIN(HistoricoMoraEmp[periodo_actualizacion]), ALL(HistoricoMoraEmp), HistoricoMoraEmp[nit] = nit, HistoricoMoraEmp[periodo_actualizacion] <= periodo)
RETURN periodo_antig
    17. Posterior

    Tabla: HistoricoMoraEmp

    
VAR nitEmp = HistoricoMoraEmp[nit]
VAR actPer = HistoricoMoraEmp[periodo_actualizacion]
VAR actYear = LEFT(actPer, 4)
VAR actMonth = RIGHT(actPer, 2)
VAR sigPer = IF(actMonth = "12", (actYear+1)*100 + 1, actYear*100 + (actMonth + 1))

RETURN sigPer
    18. Primero

    Tabla: HistoricoMoraEmp

    
VAR nitEmp = HistoricoMoraEmp[nit]
VAR ante = CONVERT(HistoricoMoraEmp[Anterior], STRING)
VAR valid = CALCULATE(MAX(HistoricoMoraEmp[periodo_actualizacion]), ALL(HistoricoMoraEmp), HistoricoMoraEmp[nit] = nitEmp, HistoricoMoraEmp[periodo_actualizacion] = ante)
RETURN IF(ISBLANK(valid), 1, 0)
    19. Ultimo

    Tabla: HistoricoMoraEmp

    
VAR nitEmp = HistoricoMoraEmp[nit]
VAR poste = CONVERT(HistoricoMoraEmp[Posterior], STRING)
VAR valid = CALCULATE(MAX(HistoricoMoraEmp[periodo_actualizacion]), ALL(HistoricoMoraEmp), HistoricoMoraEmp[nit] = nitEmp, HistoricoMoraEmp[periodo_actualizacion] = poste)
RETURN IF(ISBLANK(valid), 1, 0)
    20. Tipcategoria

    Tabla: HistoricoRecobro

    
IF(HistoricoRecobro[categoria] = "A", "Agrario",
    IF(HistoricoRecobro[categoria] = "O", "Pago incorrecto",
        IF(HistoricoRecobro[categoria] = "P", "Pensionado")
    )
)
    21. numGiro

    Tabla: MetasCuota

    
RANKX(FILTER('MetasCuota', 'MetasCuota'[Mes] = EARLIER('MetasCuota'[Mes])), 'MetasCuota'[fecha],, ASC, Dense)
    22. Ano-Cuatri

    Tabla: Periodos

    
Periodos[Año] & "-" & Periodos[Cuatrimestre]
    23. Ano-Trim

    Tabla: Periodos

    
Periodos[Año] & "-" & Periodos[Trimestre]
    24. Cuatrimestre

    Tabla: Periodos

    
VAR mes = VALUE(RIGHT(Periodos[periodo], 2))
VAR cctrim = IF( AND(1 <= mes, mes <= 4), 1, 
                 IF( AND(5 <= mes, mes <= 8), 2, 
                     IF( AND(9 <= mes, mes <= 12), 3, BLANK())
                 )
              )
RETURN cctrim
    25. Trimestre

    Tabla: Periodos

    
VAR mes = VALUE(RIGHT(Periodos[periodo], 2))
VAR cctrim = IF( AND(1 <= mes, mes <= 3), 1, 
                 IF( AND(4 <= mes, mes <= 6), 2, 
                     IF( AND(7 <= mes, mes <= 9), 3, 
                         IF( AND(10 <= mes, mes <= 12), 4, BLANK())
                     )
                 )
              )
RETURN cctrim
    26. Polaridad

    Tabla: Satisfaccion

    
IF([Calificacion] >= 4, "Satisfecho", "Insatisfecho")
    27. Medio

    Tabla: Igestion_Mercurio

    
IF(Igestion_Mercurio[Fuente] = "Mercurio_Trabajadores" || 
   Igestion_Mercurio[Fuente] = "Mercurio_Beneficiarios" || 
   Igestion_Mercurio[Fuente] = "Mercurio_Empresas", "Mercurio", "Igestion")
